## Overview

Karma POS includes a comprehensive financing system that allows you to offer credit to customers, track payment plans, manage installments, and generate payment receipts. The system supports multiple payment methods and provides detailed tracking of all financial transactions.

## How It Works

### 1. Loading Financing Records

Load all active financing records for display:

```typescript
loadFinancings(showMessage = false, number: number = 0) {
  this.posService.getFinancings().subscribe({
    next: (res) => {
      this.financings = res.data || [];
      
      // Reload selected financing if viewing payments
      if (number === 1) {
        this.selectedFinancing = this.financings.find(
          (f) => f.id === this.selectedFinancing?.id
        ) || null;
        this.viewPayments(this.selectedFinancing);
      }
      
      this.filteredFinancings = [...this.financings];
      
      if (showMessage) {
        this.messageService.add({
          severity: 'success',
          summary: 'Datos actualizados',
          detail: 'Se cargaron todas las financiaciones correctamente.'
        });
      }
    },
    error: () => {
      this.messageService.add({
        severity: 'error',
        summary: 'Error',
        detail: 'No se pudieron cargar las financiaciones.'
      });
    }
  });
}
```

#### Saving Payment

Validate and save a new payment:

```typescript
savePayment() {
  // Validate payment amount
  if (!this.newPayment.amount || this.newPayment.amount <= 0) {
    this.messageService.add({
      severity: 'warn',
      summary: 'Campo requerido',
      detail: 'Debes ingresar un valor de abono válido.'
    });
    return;
  }
  
  // Validate payment doesn't exceed remaining balance
  const remainingBalance = this.selectedFinancing?.currentBalance || 0;
  if (this.newPayment.amount > remainingBalance) {
    this.messageService.add({
      severity: 'error',
      summary: 'Monto inválido',
      detail: `El valor ingresado (${this.formatCurrency(this.newPayment.amount)}) 
               supera el saldo pendiente (${this.formatCurrency(remainingBalance)}).`
    });
    return;
  }
  
  const payload = {
    id: this.selectedFinancing.id,
    paymentAmount: this.newPayment.amount,
    paymentMethod: this.newPayment.method,
    notes: this.newPayment.notes,
    registeredByUser: 1
  };
  
  this.posService.addFinancingPayment(payload).subscribe({
    next: (res) => {
      this.messageService.add({
        severity: 'success',
        summary: 'Abono registrado',
        detail: res.message || 'Pago registrado correctamente.'
      });
      
      this.showPaymentDialog = false;
      this.loadFinancings(false, 1);
    },
    error: () => {
      this.messageService.add({
        severity: 'error',
        summary: 'Error',
        detail: 'No se pudo registrar el abono.'
      });
    }
  });
}
```

## Filtering and Search

Filter financing records by customer or invoice:

```typescript
applyFilters() {
  const c = this.filters.customer.toLowerCase().trim();
  const i = this.filters.invoice.toLowerCase().trim();
  
  this.filteredFinancings = this.financings.filter((f) => 
    (!c || f.customerName.toLowerCase().includes(c)) && 
    (!i || f.invoiceNumber.toLowerCase().includes(i))
  );
}

resetFilters() {
  this.filters = { customer: '', invoice: '' };
  this.filteredFinancings = [...this.financings];
}
```

## Permissions

The financing module respects role-based permissions:

```typescript
private applyPermissions(): void {
  const moduleName = 'Financiaciones';
  const permission = this.permissions.find(
    (p) => p.module === moduleName
  );
  
  if (!permission) {
    this.resetPermissions();
    return;
  }
  
  this.canView = permission.canView;
  this.canCreate = permission.canCreate;
  this.canEdit = permission.canEdit;
  this.canDelete = permission.canDelete;
  this.canExport = permission.canExport;
}
```
