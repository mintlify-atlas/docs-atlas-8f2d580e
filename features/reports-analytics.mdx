### Percentage Calculation

Category percentage is calculated as:

```
percentage = (category_amount / total_sales) * 100
```

For example, if Beverages generated $456.75 out of $1,247.50 total:
```
percentage = (456.75 / 1247.50) * 100 = 36.6%
```

ngOnInit() {

## Performance Optimization

For large datasets, consider:

### Data Aggregation

Pre-aggregate data on the backend:
```typescript
// Backend aggregation example
GET /api/Reports/SalesSummary?period=month&groupBy=day

// Returns pre-calculated daily totals for the month
```

### Lazy Loading

Load detailed data on demand:
```typescript
// Load summary first
loadSummary();

// Load details when user clicks
onViewDetails(category: string) {
  this.loadCategoryDetails(category);
}
```

### Caching

Cache frequently accessed reports:
```typescript
private reportCache = new Map<string, any>();

getReport(key: string) {
  if (this.reportCache.has(key)) {
    return of(this.reportCache.get(key));
  }
  
  return this.reportsService.getReport(key).pipe(
    tap(data => this.reportCache.set(key, data))
  );
}
```

## API Integration

While the current implementation uses mock data, the system is designed to integrate with backend APIs:

```typescript
// Future service implementation
@Injectable({ providedIn: 'root' })
export class ReportsService {
  private apiUrl = 'https://localhost:7197/api/';
  
  constructor(private http: HttpClient) {}
  
  getSalesSummary(period: string): Observable<SalesPeriodData> {
    return this.http.get<SalesPeriodData>(
      `${this.apiUrl}Reports/SalesSummary?period=${period}`
    );
  }
  
  getTopProducts(period: string, limit: number = 10): 
    Observable<TopProduct[]> {
    return this.http.get<TopProduct[]>(
      `${this.apiUrl}Reports/TopProducts?period=${period}&limit=${limit}`
    );
  }
  
  getCategorySales(period: string): Observable<CategorySales[]> {
    return this.http.get<CategorySales[]>(
      `${this.apiUrl}Reports/CategorySales?period=${period}`
    );
  }
  
  exportToExcel(reportType: string, filters: any): Observable<Blob> {
    return this.http.post(
      `${this.apiUrl}Reports/ExportExcel`,
      { reportType, filters },
      { responseType: 'blob' }
    );
  }
  
  exportToPDF(reportType: string, filters: any): Observable<Blob> {
    return this.http.post(
      `${this.apiUrl}Reports/ExportPDF`,
      { reportType, filters },
      { responseType: 'blob' }
    );
  }
}
```
